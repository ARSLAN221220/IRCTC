2.	List all the routes in descending order of seats sold, including route information and distribution of seats sold (Children, Adult, Senior Citizen) in the month of October this year.
QUERY ->	SELECT r.RouteID,r.SourceStation,r.DestinationStation,
			COUNT(b.TicketID) AS TotalSeatsSold,
			SUM(CASE WHEN p.Age < 12 THEN 1 ELSE 0 END) AS ChildrenSeats,
			SUM(CASE WHEN p.Age BETWEEN 12 AND 65 THEN 1 ELSE 0 END) AS AdultSeat,
			SUM(CASE WHEN p.Age > 65 THEN 1 ELSE 0 END) AS SeniorSeat
			FROM Route r
			JOIN Schedule s ON r.RouteID = s.RouteID
			LEFT JOIN ticket b ON s.ScheduleID = b.ScheduleID
			LEFT JOIN Passenger p ON b.PassengerID = p.PassengerID
			WHERE MONTH(s.DepartureDate) = 10 AND YEAR(s.DepartureDate) = YEAR(CURDATE())
			GROUP BY r.RouteID
			ORDER BY  TotalSeatsSold DESC;

3.	List all agentsâ€™ information with more than 10 confirmed bookings in the month of September this year.
QUERY -> 	SELECT a.* FROM travelagent a
			JOIN ticket b ON a.AgentID = b.AgentID
			WHERE MONTH(b.BookingDate) = 9 AND YEAR(b.BookingDate) = YEAR(CURDATE()) AND b.BookingStatus = 'Confirmed'
			GROUP BY a.AgentID
			HAVING COUNT(b.TicketID) > 10;

4.	Display the details of the route most travelled by Senior Citizens.
QUERY -> 	SELECT r.*,COUNT(p.PassengerID) AS SeniorCitizenTravelCount
			FROM Route r
			JOIN Schedule t ON r.RouteID = t.RouteID
			JOIN Ticket tk ON t.ScheduleID = tk.ScheduleID
			JOIN Passenger p ON tk.PassengerID = p.PassengerID
			WHERE p.DiscountType = 'Senior'
			GROUP BY r.RouteID
			ORDER BY SeniorCitizenTravelCount DESC;

5.	Display the details of the route where a train was always on time.
QUERY -> 	SELECT r.*,s.TrainID,MAX(s.DepartureDate) AS MaxDepartureDate,s.ArrivalDate
			FROM Route r
			JOIN Train t ON r.RouteID = t.RouteID
			JOIN Schedule s ON t.TrainID = s.TrainID
			WHERE s.OnTime = 1
			GROUP BY r.RouteID,s.TrainID,s.ArrivalDate
			HAVING COUNT(s.TrainID) = COUNT(CASE WHEN s.OnTime = 1 THEN 1 END);
